    const newCount=3-Math.floor(elapsed/1000);
    if(newCount!==count&&newCount>0){count=newCount;cdNum.textContent=count;const freqs={2:550,1:700};beep(freqs[count]||440,0.2);}
    if(elapsed>=total){clearInterval(iv);cdNum.textContent='💥';setTimeout(()=>{document.getElementById('moCountdown').style.display='none';revealCell();},400);}
  },step);
}

function revealCell(){
  const idx=pendingCell;
  const prize=S.board[idx];
  S.revealed[idx]=prize;
  const nameKey=currentUser.name.toLowerCase();
  S.players[nameKey]=true;
  const entry={cell:idx+1,name:currentUser.name,team:currentUser.team,prize:prize.n,type:prize.type,nucleo:selectedNucleo,ts:new Date().toLocaleTimeString('pt-BR')};
  S.history.unshift(entry);saveS();
  updateCellUI(idx,prize);updateStats();updateHistPanel();updateAdmPanel();
  if(brindesOpen)renderBrindeList();
  playReveal();setTimeout(()=>showPrize(prize),200);
}

}

}

function setBrindeFilter(filter,el){
  brindeFilter=filter;
  document.querySelectorAll('#brindeFilters .bfbtn').forEach(b=>b.classList.remove('active'));
  el.classList.add('active');
  renderBrindeList();

function getRevealedByIdx(){
  const map={};
  S.history.forEach(h=>{
    const idx=h.cell-1;
    if(!map[idx])map[idx]=h;
  });
  return map;
}

function renderBrindeList(){
  const list=document.getElementById('brindeList');
  const counter=document.getElementById('brindeCounter');
  if(!list||!counter)return;
  const revMap=getRevealedByIdx();

  let items=S.board.map((prize,idx)=>({prize,idx,isRevealed:!!S.revealed[idx],revEntry:revMap[idx]||null}));

  if(brindeFilter==='hidden'){
    items=items.filter(it=>!it.isRevealed);
  }else if(brindeFilter!=='all'){
    items=items.filter(it=>it.prize.type===brindeFilter);
  }

  counter.textContent='Exibindo '+items.length+' de 80 itens';

  if(items.length===0){
    list.innerHTML='<div style="padding:14px;text-align:center;color:var(--text3);font-size:10px">Nenhum item encontrado.</div>';
    return;
  }

  list.innerHTML=items.map(it=>{
    const revClass=it.isRevealed?'rev-'+it.prize.type:'';
    const badgeClass=it.isRevealed?'revealed-'+it.prize.type:'hidden';
    const badgeText=it.isRevealed?'✅ REVELADO':'🔒 OCULTO';
    const playerInfo=it.revEntry?'<div class="brinde-player">▸ '+escHtml(it.revEntry.name)+'</div>':'';
    return '<div class="brinde-item '+revClass+'" onclick="flashCell('+it.idx+')" title="Clique para localizar no tabuleiro">'
      +'<span class="brinde-cell-num">#'+(it.idx+1)+'</span>'
      +'<span class="brinde-icon">'+it.prize.i+'</span>'
      +'<div class="brinde-info">'
        +'<div class="brinde-name">'+escHtml(it.prize.n)+'</div>'
        +playerInfo
      +'</div>'
      +'<span class="brinde-badge '+badgeClass+'">'+badgeText+'</span>'
      +'</div>';
  }).join('');
}

function escHtml(str){
  return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function flashCell(idx){
  const cell=document.getElementById('cell'+idx);
  if(!cell)return;
  cell.scrollIntoView({behavior:'smooth',block:'center'});
  cell.classList.remove('cell-flash-anim');
  void cell.offsetWidth;
  cell.classList.add('cell-flash-anim');
  setTimeout(()=>{
    cell.classList.remove('cell-flash-anim');
  },1100);
}

function exportCSV(){
  const rows=[['Célula','Nome','Equipe','Prêmio','Tipo','Núcleo','Horário']];
  S.history.forEach(h=>rows.push([h.cell,h.name,h.team,h.prize,h.type,h.nucleo||'',h.ts]));
  const csv=rows.map(r=>r.map(v=>'"'+String(v).replace(/"/g,'""')+'"').join(',')).join('\n');
  const a=document.createElement('a');
  a.href='data:text/csv;charset=utf-8,\uFEFF'+encodeURIComponent(csv);
  a.download='ocmg_historico.csv';
  a.click();
}
